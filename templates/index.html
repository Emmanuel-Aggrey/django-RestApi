simple example
Get All Data by requests Generated by POSTMAN
import requests
url = "http://127.0.0.1:8000/api/persons/"
payload = {}
headers = { 'Authorization': 'Token cd3f275374f2441ca7208fafb9056b34a703f9e6',
'Authorization': 'Basic ZW1tYW46YWdncmV5MTIz'
}
response = requests.request("GET", url, headers=headers, data = payload)
print(response.text.encode('utf8'))

Get all Data by curl Generated by POSTMAN
curl --location --request GET 'http://127.0.0.1:8000/api/persons/' \
--header 'Authorization: Token cd3f275374f2441ca7208fafb9056b34a703f9e6' \
--header 'Authorization: Basic ZW1tYW46YWdncmV5MTIz'
